const express = require("express");
const mysql = require("mysql2/promise");

const app = express();
app.use(express.json());

// 🔌 MySQL bağlantısı
const pool = mysql.createPool({
  host: "localhost",
  user: "cursor_dev",
  password: "test123",
  database: "myapp_test",
  waitForConnections: true,
  connectionLimit: 10,
});

// ⚙️ DRY RUN MODE
const DRY_RUN = true;

/**
 * TOOL: Create Table
 * body:
 * {
 *   table: "users",
 *   columns: [
 *     { name: "id", type: "INT AUTO_INCREMENT PRIMARY KEY" },
 *     { name: "email", type: "VARCHAR(255)" }
 *   ]
 * }
 */
app.post("/create-table", async (req, res) => {
  try {
    const { table, columns } = req.body;

    if (!table || !columns || !Array.isArray(columns)) {
      return res.status(400).json({ error: "Invalid payload" });
    }

    const colsSQL = columns
      .map(col => `\`${col.name}\` ${col.type}`)
      .join(", ");

    const query = `CREATE TABLE \`${table}\` (${colsSQL});`;

    if (DRY_RUN) {
      return res.json({
        dryRun: true,
        query,
        message: "No changes executed (dry run enabled)"
      });
    }

    const conn = await pool.getConnection();
    await conn.query(query);
    conn.release();

    res.json({ ok: true, query });
  } catch (err) {
    console.error(err);
    res.status(500).json({ error: err.message });
  }
});

/**
 * TOOL: Add Column
 * body:
 * {
 *   table: "users",
 *   column: "age",
 *   type: "INT"
 * }
 */
app.post("/add-column", async (req, res) => {
  try {
    const { table, column, type } = req.body;

    if (!table || !column || !type) {
      return res.status(400).json({ error: "Invalid payload" });
    }

    const query = `ALTER TABLE \`${table}\` ADD COLUMN \`${column}\` ${type};`;

    if (DRY_RUN) {
      return res.json({
        dryRun: true,
        query,
        message: "No changes executed (dry run enabled)"
      });
    }

    const conn = await pool.getConnection();
    await conn.query(query);
    conn.release();

    res.json({ ok: true, query });
  } catch (err) {
    console.error(err);
    res.status(500).json({ error: err.message });
  }
});

/**
 * TOOL: List Tables (Cursor için çok faydalı)
 */
app.get("/tables", async (req, res) => {
  try {
    const [rows] = await pool.query("SHOW TABLES;");
    res.json(rows);
  } catch (err) {
    res.status(500).json({ error: err.message });
  }
});

// 🚀 start server
app.listen(3001, () => {
  console.log("MCP MySQL server running on http://localhost:3001");
  console.log("DRY_RUN =", DRY_RUN);
});
